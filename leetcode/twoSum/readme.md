- 数组
var arr = [1, 2, 3];
arr.unshift(0); // [0, 1, 2, 3]
arr.shift(); // [1, 2, 3, 4]
arr.push(4); // [0, 1, 2, 3, 4]
arr.pop(); // [1, 2, 3]
arr.splice(1, 2); // [1, 3]
